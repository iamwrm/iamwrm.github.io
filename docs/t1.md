    #用来计数阶乘数的末尾0的函数
          count=0;
          result=fact(x)
          while result % 10==0:
              count+=1
              result/=10
          return count
    
    for i in range(10,20,1):

### 高阶函数

将一个函数作为一个函数的参数，后者就是高阶函数

    def f1(x):
        return x*x
    def f2(x):
        return x**3
    
    def t1(a,b,f):
        return abs(f(b)-f(a))
    
    print(t1(1,4,f1))
    print(t1(1,4,f2))

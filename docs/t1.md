1.  use filter to find prime numbers

        def _odd_iter():
            n = 1
            while True:
                n = n + 2
                yield n
        
        def _not_divisible(n):
            return lambda x: x % n > 0
        
        def primes():
            yield 2
            it = _odd_iter() # 初始序列
            while True:
                n = next(it) # 返回序列的第一个数
                yield n
                it = filter(_not_divisible(n), it) # 构造新序列
        
        # 打印100以内的素数:
        for n in primes():
            if n < 100:
                print(n,end=" ")
            else:
                break

        def by_score(t):
            (a,b)=t

            (a,b)=t
            return a
        
        print(list(sorted(L,key=by_name)))
